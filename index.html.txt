/* ============ STORAGE HELPERS ============ */
function loadAddedProducts(){
  try{
    const raw = localStorage.getItem(STORAGE_KEY);
    return raw ? JSON.parse(raw) : [];
  }catch(e){
    console.warn("Could not read saved products:", e);
    return [];
  }
}
function saveAddedProducts(list){
  try{
    localStorage.setItem(STORAGE_KEY, JSON.stringify(list));
    return true;
  }catch(e){
    console.warn("Could not save products:", e);
    return false;
  }
}
function slug(str){
  return str.toLowerCase().replace(/[^a-z0-9]+/g,'-').replace(/(^-|-$)/g,'');
}

/* ============ RENDER STOREFRONT ============ */
function getAllProducts(){
  return [...SEED_PRODUCTS, ...loadAddedProducts()];
}

function renderNav(){
  const navEl = document.getElementById('directory-nav');
  const heroList = document.getElementById('hero-directory');
  navEl.innerHTML = CATEGORIES.map(c => `<a href="#${slug(c)}">${c}</a>`).join('');
  heroList.innerHTML = CATEGORIES.map(c => `<li><a href="#${slug(c)}">${c}</a></li>`).join('');
}

function renderRooms(){
  const container = document.getElementById('rooms-container');
  const all = getAllProducts();
  container.innerHTML = CATEGORIES.map(cat => {
    const items = all.filter(p => p.category === cat);
    const cards = items.length ? items.map(cardHtml).join('') :
      `<div class="empty-note">No products in this section yet — check back soon.</div>`;
    return `
      <div class="shelf"><div class="line"></div></div>
      <section class="room" id="${slug(cat)}">
        <div class="room-head">
          <h2>${cat}</h2>
          <div class="count">${items.length} ${items.length === 1 ? 'piece' : 'pieces'}</div>
        </div>
        <div class="grid">${cards}</div>
      </section>
    `;
  }).join('');
}

function cardHtml(p){
  return `
    <div class="card">
      <div class="thumb"><img src="${escapeHtml(p.imageUrl)}" alt="${escapeHtml(p.name)}" loading="lazy"></div>
      <div class="body">
        <div class="name">${escapeHtml(p.name)}</div>
        <div class="price">${escapeHtml(p.price)}</div>
        ${p.note ? `<div class="note">${escapeHtml(p.note)}</div>` : ''}
        <a class="cta" href="${escapeHtml(p.link)}" target="_blank" rel="noopener noreferrer nofollow">View item</a>
      </div>
    </div>
  `;
}

function escapeHtml(str){
  const d = document.createElement('div');
  d.textContent = str == null ? '' : String(str);
  return d.innerHTML;
}

function renderAdminList(){
  const list = loadAddedProducts();
  document.getElementById('added-count').textContent = list.length;
  const wrap = document.getElementById('added-products');
  if(!list.length){
    wrap.innerHTML = `<div style="opacity:0.6; font-size:0.85rem; margin-top:10px;">Nothing added yet.</div>`;
    return;
  }
  wrap.innerHTML = list.map(p => `
    <div class="admin-row" data-id="${p.id}">
      <img src="${escapeHtml(p.imageUrl)}" alt="">
      <div class="meta">
        <div class="r-cat">${escapeHtml(p.category)}</div>
        <div>${escapeHtml(p.name)} — ${escapeHtml(p.price)}</div>
      </div>

    </div>
  `).join('');

      renderRooms();
    });
  });
}
